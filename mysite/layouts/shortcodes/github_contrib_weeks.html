{{- $username := .Get "username" | default "chriskersov" -}} {{- $token := getenv "HUGO_PARAMS_GITHUB_TOKEN" -}} {{- $headers := dict "Content-Type" "application/json" -}} {{- if $token -}} {{- $headers = merge $headers (dict "Authorization" (printf "bearer %s" $token)) -}} {{- end -}} {{/* Construct the GraphQL Query */}} {{- $q1 := printf "query { user(login: %q) {" $username -}} {{- $q2 := " contributionsCollection { contributionCalendar {" -}} {{- $q3 := " weeks { contributionDays { contributionCount date } } } } } }" -}} {{- $query := printf "%s%s%s" $q1 $q2 $q3 -}} {{/* Set up Request Options */}} {{- $body := dict "query" $query | jsonify -}} {{/* Create a unique ID based on the current hour/day to bust the cache */}} {{- $cacheBuster := now.Format "2006-01-02-15" -}} {{- $opts := dict "method" "post" "body" $body "headers" $headers "key" (printf "%s-%s" $username $cacheBuster) -}} {{- $weeks := slice -}} {{/* Fetch Data from GitHub API */}} {{- with try (resources.GetRemote "https://api.github.com/graphql" $opts) -}} {{- with .Value -}} {{- $payload := . | transform.Unmarshal -}} {{- with index $payload "data" "user" "contributionsCollection" "contributionCalendar" "weeks" -}} {{- $weeks = . -}} {{- end -}} {{- end -}} {{- end -}} {{/* Output the Result as JSON */}} {{- $weeks | jsonify -}}
